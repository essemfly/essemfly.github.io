# bisect

Time Complexity: O(log n)


import bisect

bisect.bisect_left(a, x)

>>> x = [1, 5, 9, 2, 4, 8]
>>> y = 3
>>> bisect.bisect_left(x, y)

4

>>> x.sort()
>>> bisect.bisect_left(x, y)

2

>>> x

[1, 2, 4, 5, 8, 9]
